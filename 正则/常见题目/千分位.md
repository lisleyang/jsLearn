> 题目：将1234567 变成 1,234,567，即千分位标注

```javascript
var str = '12345678';
var arr = str.split('') //Array.prototype.slice.call(str,'');
var resultArr = [];
for(var i=arr.length-1;i>=0;i--){
    if(((arr.length-1-i)%3 == 0) &&(arr.length-1-i>0)){
        resultArr.unshift(',');
    }
    resultArr.unshift(arr[i]);
}
console.log(resultArr.join(''));
```
